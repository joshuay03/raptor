Overview

Per-connection outbound flow-control accounting.

Tracks the peer's connection-level and per-stream receive windows so outbound DATA frames respect RFC 7540 §5.2. Threads dispatching stream responses call acquire to reserve send capacity; threads applying inbound WINDOW_UPDATE or SETTINGS frames call the mutating methods to replenish it. State is held in a single Atom so updates use CAS.

def acquire(stream_id, max_bytes)
  loop do
    granted = 0
    @state.swap do |state|
      stream_window = state[:streams][stream_id] || state[:initial_stream]
      granted = [state[:connection], stream_window, max_bytes, MAX_FRAME_SIZE].min
      if granted > 0
        state.merge(
          connection: state[:connection] - granted,
          streams: state[:streams].merge(stream_id => stream_window - granted)
        ).freeze
      else
        state
      end
    end

    return granted if granted > 0

    sleep ACQUIRE_POLL_INTERVAL
  end
end

def add_stream_window(stream_id, increment)
  @state.swap do |state|
    current = state[:streams][stream_id] || state[:initial_stream]
    state.merge(streams: state[:streams].merge(stream_id => current + increment)).freeze
  end
end

def set_initial_stream_window(new_size)
  @state.swap do |state|
    delta = new_size - state[:initial_stream]
    shifted = state[:streams].transform_values { |size| size + delta }
    state.merge(initial_stream: new_size, streams: shifted).freeze
  end
end
